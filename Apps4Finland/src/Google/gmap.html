<script type="text/javascript"

<script type="text/javascript">
      function initialize() {
        var mapOptions = {
          center: new google.maps.LatLng(50.8419, 4.3773),
          zoom: 8,
          mapTypeId: google.maps.MapTypeId.ROADMAP
        };
        var map = new google.maps.Map(document.getElementById("map"),
            mapOptions);
      }
    </script>
